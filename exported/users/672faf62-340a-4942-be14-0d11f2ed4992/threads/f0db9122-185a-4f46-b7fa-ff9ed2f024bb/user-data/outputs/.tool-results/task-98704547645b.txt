import { useState, useEffect } from 'react';
import Taro, { useRouter } from '@tarojs/taro';
import { View, Text, ScrollView } from '@tarojs/components';
import { api } from '../../services/api';
import './index.scss';

interface Profession {
  id: number;
  name: string;
  match_percentage: number;
  description?: string;
  category?: string;
}

type MedalColor = '#FFD700' | '#C0C0C0' | '#CD7F32' | '';

function getMedal(index: number): { color: MedalColor; emoji: string } {
  if (index === 0) return { color: '#FFD700', emoji: '🥇' };
  if (index === 1) return { color: '#C0C0C0', emoji: '🥈' };
  if (index === 2) return { color: '#CD7F32', emoji: '🥉' };
  return { color: '', emoji: '' };
}

export default function Profession() {
  const router = useRouter();
  const { id } = router.params;
  const [professions, setProfessions] = useState<Profession[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState('');

  useEffect(() => {
    loadData();
  }, [id]);

  const loadData = async () => {
    setLoading(true);
    setError('');
    try {
      let data: Profession[];
      if (id) {
        const result = await api.getResult(Number(id));
        data = result.professions || [];
      } else {
        data = await api.getProfessions();
      }
      setProfessions(data.sort((a, b) => b.match_percentage - a.match_percentage));
    } catch (e: any) {
      setError(e.message || '加载失败');
      Taro.showToast({ title: '加载失败', icon: 'none' });
    } finally {
      setLoading(false);
    }
  };

  const barColor = (pct: number): string => {
    if (pct >= 80) return 'linear-gradient(90deg, #07c160, #06ad56)';
    if (pct >= 60) return 'linear-gradient(90deg, #667eea, #5a6fd6)';
    if (pct >= 40) return 'linear-gradient(90deg, #f39c12, #e67e22)';
    return 'linear-gradient(90deg, #ccc, #b0b0b0)';
  };

  if (loading) {
    return (
      <View className="loading-wrap">
        <Text className="loading-text">加载中...</Text>
      </View>
    );
  }

  if (error) {
    return (
      <View className="error-wrap">
        <Text className="error-text">{error}</Text>
        <View className="retry-btn" onClick={loadData}>
          <Text>重新加载</Text>
        </View>
      </View>
    );
  }

  return (
    <View className="profession-page">
      <View className="header">
        <Text className="header-title">专业匹配排名</Text>
        <Text className="header-subtitle">匹配度越高，越适合你</Text>
      </View>

      <ScrollView scrollY className="rank-list">
        {professions.map((prof, index) => {
          const medal = getMedal(index);
          const isTop3 = index < 3;

          return (
            <View
              key={prof.id}
              className={`rank-item ${isTop3 ? 'top-rank' : ''}`}
            >
              <View className="rank-badge">
                {isTop3 ? (
                  <Text className="medal-emoji">{medal.emoji}</Text>
                ) : (
                  <Text className="rank-number">{index + 1}</Text>
                )}
              </View>

              <View className="rank-body">
                <View className="rank-top-row">
                  <Text className="prof-name">{prof.name}</Text>
                  <Text
                    className="match-pct"
                    style={{ color: medal.color || '#333' }}
                  >
                    {prof.match_percentage}%
                  </Text>
                </View>

                <View className="bar-track">
                  <View
                    className="bar-fill"
                    style={{
                      width: `${prof.match_percentage}%`,
                      background: barColor(prof.match_percentage),
                    }}
                  />
                </View>

                {prof.description && (
                  <Text className="prof-desc">{prof.description}</Text>
                )}
              </View>
            </View>
          );
        })}
      </ScrollView>
    </View>
  );
}
</write>

import Taro from '@tarojs/taro';
import { API_BASE_URL } from '../utils/constants';
import { getToken } from './auth';
import { Scale, Question, Result } from '../types';

function request<T>(url: string, options: any = {}): Promise<T> {
  const token = getToken();
  const header: any = { 'Content-Type': 'application/json' };
  if (token) header['Authorization'] = `Bearer ${token}`;

  return new Promise((resolve, reject) => {
    Taro.request({
      url: `${API_BASE_URL}${url}`,
      header,
      ...options,
      success(res) {
        if (res.statusCode === 200) resolve(res.data as T);
        else if (res.statusCode === 401) {
          Taro.reLaunch({ url: '/pages/login/index' });
          reject(new Error('未登录'));
        } else reject(new Error(res.data?.detail || '请求失败'));
      },
      fail(err) {
        Taro.showToast({ title: '网络错误', icon: 'none' });
        reject(err);
      },
    });
  });
}

export const api = {
  login: (code: string) =>
    request<any>('/auth/login', { method: 'POST', data: { code } }),

  getScales: () => request<Scale[]>('/scales'),

  getScaleDetail: (id: number) => request<Scale>(`/scales/${id}`),

  getQuestions: (id: number) => request<Question[]>(`/scales/${id}/questions`),

  submitAnswers: (
    scaleId: number,
    answers: { question_id: number; value: number }[]
  ) =>
    request<Result>(`/scales/${scaleId}/submit`, {
      method: 'POST',
      data: { answers },
    }),

  getResult: (id: number) => request<Result>(`/results/${id}`),

  getResults: () => request<Result[]>('/results'),

  getProfessions: () => request<any[]>('/professions'),
};

<write file="/mnt/user-data/workspace/scale-miniapp/src/services/auth.ts">
import Taro from '@tarojs/taro';
import { api } from './api';

const TOKEN_KEY = 'auth_token';
const USER_KEY = 'auth_user';

export function getToken(): string | null {
  return Taro.getStorageSync(TOKEN_KEY) || null;
}

export function setToken(token: string): void {
  Taro.setStorageSync(TOKEN_KEY, token);
}

export function clearToken(): void {
  Taro.removeStorageSync(TOKEN_KEY);
  Taro.removeStorageSync(USER_KEY);
}

export function isLoggedIn(): boolean {
  return !!getToken();
}

export function getUserInfo(): any {
  const raw = Taro.getStorageSync(USER_KEY);
  return raw ? JSON.parse(raw) : null;
}

export function setUserInfo(info: any): void {
  Taro.setStorageSync(USER_KEY, JSON.stringify(info));
}

export async function wxLogin(): Promise<void> {
  const loginRes = await Taro.login();
  if (!loginRes.code) {
    throw new Error('获取微信登录凭证失败');
  }

  const data = await api.login(loginRes.code);
  if (data.token) {
    setToken(data.token);
    if (data.user) {
      setUserInfo(data.user);
    }
  } else {
    throw new Error('登录失败：服务器未返回 token');
  }
}

export function logout(): void {
  clearToken();
  Taro.reLaunch({ url: '/pages/login/index' });
}
</write>
