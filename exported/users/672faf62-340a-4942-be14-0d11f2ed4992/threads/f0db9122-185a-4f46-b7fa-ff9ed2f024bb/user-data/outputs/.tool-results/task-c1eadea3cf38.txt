import Taro from '@tarojs/taro';
import { API_BASE_URL } from '../utils/constants';
import { getToken } from './auth';
function request<T>(url:string,options:any={}):Promise<T>{
  const token=getToken();
  const header:any={'Content-Type':'application/json'};
  if(token) header['Authorization']=`Bearer ${token}`;
  return new Promise((resolve,reject)=>{
    Taro.request({url:`${API_BASE_URL}${url}`,header,...options,
      success(res){
        if(res.statusCode===200) resolve(res.data as T);
        else if(res.statusCode===401){Taro.reLaunch({url:'/pages/login/index'});reject(new Error('未登录'));}
        else reject(new Error((res.data as any)?.detail||'请求失败'));
      },
      fail(err){Taro.showToast({title:'网络错误',icon:'none'});reject(err);},
    });
  });
}
export const api={
  login:(code:string)=>request<any>('/auth/login',{method:'POST',data:{code}}),
  getScales:()=>request<any[]>('/scales'),
  getScaleDetail:(id:number)=>request<any>(`/scales/${id}`),
  getQuestions:(id:number)=>request<any[]>(`/scales/${id}/questions`),
  submitAnswers:(scaleId:number,answers:{question_id:number,value:number}[])=>request<any>(`/scales/${scaleId}/submit`,{method:'POST',data:{answers}}),
  getResult:(id:number)=>request<any>(`/results/${id}`),
  getResults:()=>request<any[]>('/results'),
};
EOF

import { useState, useEffect } from 'react';
import Taro, { useRouter } from '@tarojs/taro';
import { View, Text, Button } from '@tarojs/components';
import { api } from '../../services/api';
import { Question } from '../../types';
import { SCALE_OPTIONS, SCALE_TYPE_LABELS } from '../../utils/constants';
import './index.scss';

export default function Quiz(){
  const router = useRouter();
  const { scaleId, type } = router.params;
  const [questions, setQuestions] = useState<Question[]>([]);
  const [currentIdx, setCurrentIdx] = useState(0);
  const [answers, setAnswers] = useState<Record<number,number>>({});
  const [submitting, setSubmitting] = useState(false);

  useEffect(()=>{
    if(scaleId) loadQuestions(Number(scaleId));
  },[scaleId]);

  const loadQuestions=async(id:number)=>{
    try{
      const data=await api.getQuestions(id);
      setQuestions(data);
      const saved=Taro.getStorageSync(`quiz_${id}`);
      if(saved){ setAnswers(JSON.parse(saved)); }
    }catch(e){ Taro.showToast({title:'加载题目失败',icon:'none'}); }
  };

  const selectOption=(qId:number,value:number)=>{
    const newAnswers={...answers,[qId]:value};
    setAnswers(newAnswers);
    Taro.setStorageSync(`quiz_${scaleId}`,JSON.stringify(newAnswers));
  };

  const q=questions[currentIdx];
  const selected= q ? answers[q.id] : undefined;
  const allAnswered=questions.length>0 && questions.every(q=>answers[q.id]!==undefined);

  const handleSubmit=async()=>{
    if(!allAnswered) return;
    setSubmitting(true);
    try{
      const ansList=Object.entries(answers).map(([qId,value])=>({question_id:Number(qId),value}));
      const result=await api.submitAnswers(Number(scaleId),ansList);
      Taro.removeStorageSync(`quiz_${scaleId}`);
      Taro.redirectTo({url:`/pages/result/index?id=${result.id}`});
    }catch(e){ Taro.showToast({title:'提交失败',icon:'none'}); }
    finally{ setSubmitting(false); }
  };

  if(!q) return <View className="quiz-page"><View className="quiz-body"><Text>加载中...</Text></View></View>;

  return (
    <View className="quiz-page">
      <View className="quiz-header">
        <Text className="progress-text">{currentIdx+1} / {questions.length}</Text>
        <View className="progress-bar">
          <View className="progress-fill" style={{width:`${((currentIdx+1)/questions.length)*100}%`}} />
        </View>
      </View>
      <View className="quiz-body">
        <Text className="question-stem">{q.question_number}. {q.stem}</Text>
        <View className="options-list">
          {SCALE_OPTIONS.map(opt=>(
            <View key={opt.label} className={`option-item ${selected===opt.value?'selected':''}`}
              onClick={()=>selectOption(q.id,opt.value)}>
              <Text className="option-label">{opt.label}.</Text>
              <Text>{opt.text}</Text>
            </View>
          ))}
        </View>
      </View>
      <View className="quiz-footer">
        <Button className="btn-nav btn-prev" disabled={currentIdx===0}
          onClick={()=>setCurrentIdx(i=>i-1)}>上一题</Button>
        {currentIdx < questions.length-1 ? (
          <Button className="btn-nav btn-next" disabled={!selected}
            onClick={()=>setCurrentIdx(i=>i+1)}>下一题</Button>
        ) : (
          <Button className="btn-nav btn-submit" disabled={!allAnswered || submitting}
            loading={submitting} onClick={handleSubmit}>
            {submitting?'提交中...':'提交答卷'}
          </Button>
        )}
      </View>
    </View>
  );
}
