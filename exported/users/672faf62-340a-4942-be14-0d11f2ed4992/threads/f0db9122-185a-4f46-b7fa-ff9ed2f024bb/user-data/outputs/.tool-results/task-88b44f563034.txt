from typing import Dict, List

class ScoringEngine:
    """5点李克特量表评分引擎"""
    
    OPTION_VALUES = {"A": 5, "B": 4, "C": 3, "D": 2, "E": 1}
    
    def __init__(self, questions: List[dict], dimensions: List[dict]):
        self.questions = {q["id"]: q for q in questions}
        self.dimensions = dimensions
        
    def calculate(self, answers: Dict[int, int]) -> Dict[int, float]:
        """计算所有维度的得分，返回 {dimension_id: score_0_to_100}"""
        dim_scores = {}
        dim_counts = {}
        
        for q_id, value in answers.items():
            q = self.questions.get(q_id)
            if not q or not q.get("dimension_id"):
                continue
            
            dim_id = q["dimension_id"]
            if dim_id not in dim_scores:
                dim_scores[dim_id] = 0
                dim_counts[dim_id] = 0
            
            if q.get("is_reverse"):
                value = 6 - value
            
            dim_scores[dim_id] += value
            dim_counts[dim_id] += 1
        
        result = {}
        for dim_id, total in dim_scores.items():
            count = dim_counts[dim_id]
            max_score = count * 5
            result[dim_id] = round(total / max_score * 100, 1) if max_score > 0 else 0
        
        return result
    
    def rollup_to_parents(self, dim_scores: Dict[int, float], dimensions: List[dict]) -> Dict[int, float]:
        """将子维度得分汇总到父维度"""
        dim_map = {d["id"]: d for d in dimensions}
        parent_scores = {}
        parent_counts = {}
        
        for dim_id, score in dim_scores.items():
            dim = dim_map.get(dim_id)
            if not dim or not dim.get("parent_id"):
                continue
            
            parent_id = dim["parent_id"]
            if parent_id not in parent_scores:
                parent_scores[parent_id] = 0
                parent_counts[parent_id] = 0
            
            parent_scores[parent_id] += score
            parent_counts[parent_id] += 1
        
        result = {}
        for parent_id, total in parent_scores.items():
            result[parent_id] = round(total / parent_counts[parent_id], 1)
        
        return result

import math
from typing import Dict, List, Tuple

class ProfessionMatcher:
    """专业匹配引擎 - 所有权重从后台配置获取"""
    
    def match(self, user_scores: Dict[str, float], professions: List[dict]) -> List[Tuple[dict, float]]:
        """计算用户与各专业的余弦相似度匹配"""
        results = []
        
        for prof in professions:
            weights = prof.get("dimension_weights", {})
            dot_product = 0
            prof_norm = 0
            user_norm = 0
            
            for dim_key, weight in weights.items():
                user_val = user_scores.get(dim_key, 0) / 100.0
                dot_product += user_val * weight
                prof_norm += weight ** 2
                user_norm += user_val ** 2
            
            prof_norm = math.sqrt(prof_norm)
            user_norm = math.sqrt(user_norm)
            
            if prof_norm == 0 or user_norm == 0:
                similarity = 0
            else:
                similarity = dot_product / (prof_norm * user_norm)
            
            match_pct = round(max(0, similarity) * 100, 1)
            results.append((prof, match_pct))
        
        results.sort(key=lambda x: x[1], reverse=True)
        return results

from typing import Dict, List

class ResponsibilityAnalyzer:
    """工作职责分析"""
    
    def analyze(self, user_scores: Dict[str, float], top_professions: List[dict]) -> dict:
        strengths = [k for k, v in user_scores.items() if v > 70]
        weaknesses = [k for k, v in user_scores.items() if v < 40]
        
        duties = []
        for prof in top_professions[:5]:
            resp = prof.get("responsibilities", [])
            if isinstance(resp, list):
                duties.extend(resp)
        
        advice = self._generate_advice(strengths, weaknesses, top_professions[:3])
        
        return {
            "strengths": strengths,
            "weaknesses": weaknesses,
            "matched_duties": list(set(duties)),
            "career_advice": advice,

from app.config import settings
from fastapi import Depends

router = APIRouter(prefix="/auth", tags=["auth"])

class LoginRequest(BaseModel):
    code: str

@router.post("/login")
async def wx_login(req: LoginRequest, db: AsyncSession = Depends(get_db)):
    # For development: mock WeChat API if no credentials configured
    if not settings.WX_APPID or not settings.WX_SECRET:
        openid = f"dev_{req.code}"
    else:
        async with httpx.AsyncClient() as client:
            resp = await client.get(
                "https://api.weixin.qq.com/sns/jscode2session",
                params={
                    "appid": settings.WX_APPID,
                    "secret": settings.WX_SECRET,
                    "js_code": req.code,
                    "grant_type": "authorization_code",
                }
            )
            wx_data = resp.json()
            if "errcode" in wx_data and wx_data["errcode"] != 0:
                raise HTTPException(400, detail=f"微信登录失败: {wx_data.get('errmsg')}")
            openid = wx_data["openid"]
    
    result = await db.execute(select(User).where(User.openid == openid))
    user = result.scalar_one_or_none()
    
    if not user:
        user = User(openid=openid, nickname=f"用户{openid[-6:]}")
        db.add(user)
        await db.flush()
    
    token = jwt.encode(
        {"user_id": user.id, "openid": openid, "exp": datetime.utcnow() + timedelta(minutes=settings.JWT_EXPIRE_MINUTES)},
        settings.JWT_SECRET,
        algorithm=settings.JWT_ALGORITHM,
    )
    
    return {
        "token": token,
        "user": {"id": user.id, "nickname": user.nickname, "avatar_url": user.avatar_url},
    }

@router.post("/{scale_id}/submit")
async def submit_answers(scale_id: int, body: dict, user=Depends(get_current_user), db: AsyncSession = Depends(get_db)):
    answers_list = body.get("answers", [])
    answers = {a["question_id"]: a["value"] for a in answers_list}
    
    # Get questions with dimensions
    q_result = await db.execute(
        select(Question).where(Question.scale_id == scale_id)
    )
    questions = q_result.scalars().all()
    q_dicts = [{"id": q.id, "dimension_id": q.dimension_id, "is_reverse": q.is_reverse} for q in questions]
    
    # Get dimensions
    d_result = await db.execute(
        select(ScaleDimension).where(ScaleDimension.scale_id == scale_id)
    )
    dimensions = d_result.scalars().all()
    d_dicts = [{"id": d.id, "code": d.code, "name": d.name, "parent_id": d.parent_id, "level": d.level} for d in dimensions]
    
    # Score
    engine = ScoringEngine(q_dicts, d_dicts)
    dim_scores = engine.calculate(answers)
    
    # Roll up to parent dimensions (大维度)
    parent_scores = engine.rollup_to_parents(dim_scores, d_dicts)
    
    # Build dimension key map for profession matching
    dim_map = {d.id: f"{'天赋' if d.level else '喜欢'}_{d.code}" for d in d_dicts}
    score_map = {}
    for dim_id, score in {**dim_scores, **parent_scores}.items():
        key = dim_map.get(dim_id, str(dim_id))
        score_map[key] = score
    
    # Match professions
    p_result = await db.execute(select(Profession))
    professions = p_result.scalars().all()
    p_dicts = [{"id": p.id, "name": p.name, "category": p.category,
                "dimension_weights": p.dimension_weights, "description": p.description,
                "responsibilities": p.responsibilities} for p in professions]
    
    matcher = ProfessionMatcher()
    ranking = matcher.match(score_map, p_dicts)
    
    # Analyze
    analyzer = ResponsibilityAnalyzer()
    analysis = analyzer.analyze(score_map, [p for p, _ in ranking])
    
    # Save result
    result = Result(
        user_id=user["user_id"], scale_id=scale_id,
        answers=answers, dimension_scores=score_map,
        profession_ranking=[{"profession_id": p["id"], "profession_name": p["name"],
                            "match_percentage": s} for p, s in ranking],

    )

    await db.flush()
    
    return {
        "id": result.id, "user_id": result.user_id, "scale_id": result.scale_id,
        "dimension_scores": score_map,
        "profession_ranking": [{"profession_id": p["id"], "profession_name": p["name"],
                               "category": p.get("category"), "match_percentage": s,
                               "description": p.get("description"),
                               "responsibilities": p.get("responsibilities")} for p, s in ranking],
        "responsibility_analysis": analysis,
