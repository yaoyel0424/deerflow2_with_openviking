### 1.3 微信登录流程

```typescript
// services/auth.ts - 微信登录实现
import Taro from '@tarojs/taro';

export async function wxLogin(): Promise<LoginResult> {
  // 1. 调用 wx.login 获取 code
  const { code } = await Taro.login();
  
  // 2. 发送 code 到后端换取 token
  const res = await Taro.request({
    url: 'https://api.example.com/auth/login',
    method: 'POST',
    data: { code }
  });
  
  // 3. 存储 token
  const { token, userInfo } = res.data;
  Taro.setStorageSync('token', token);
  Taro.setStorageSync('userInfo', userInfo);
  
  return { token, userInfo };
}
```

**关键设计要点：**
- 使用 `Taro.login()` 获取临时 code，发送到后端换取自定义 token
- 如需用户头像昵称，使用 `<Button open-type="getUserInfo">` 引导授权
- token 过期后自动刷新或重新登录
- 支持静默登录（不弹授权框，仅获取 openid）

import httpx

router = APIRouter(prefix="/auth", tags=["认证"])

@router.post("/login")
async def wx_login(code: str):
    """微信小程序登录，code 换 openid 和 session_key"""

    async with httpx.AsyncClient() as client:

### 2.3 评分引擎

评分引擎是整个系统的核心，需要灵活支持不同的量表计分规则。

```python
# services/scoring.py
from enum import Enum
from dataclasses import dataclass
from typing import Dict, List, Optional

class ScoreMode(str, Enum):
    """计分模式"""
    DIRECT = "direct"          # 直接计分（选项值即分数）
    REVERSE = "reverse"        # 反向计分
    DIMENSION = "dimension"    # 按维度汇总计分

@dataclass
class ScoringRule:
    """评分规则定义"""
    dimension: str             # 维度名称（如：逻辑天赋、艺术天赋）
    question_ids: List[int]    # 该维度包含的题目 ID
    reverse_ids: List[int]     # 该维度中需要反向计分的题目 ID
    weight: float = 1.0        # 维度权重

class ScoringEngine:
    """量表评分引擎"""
    
    def __init__(self, rules: List[ScoringRule]):
        self.rules = rules
    
    def calculate(self, answers: Dict[int, int]) -> Dict[str, float]:
        """
        根据答题结果计算各维度得分
        
        Args:
            answers: {question_id: selected_option_value}
        
        Returns:
            {dimension_name: score}
        """
        scores = {}
        for rule in self.rules:
            dim_score = 0
            for qid in rule.question_ids:
                if qid in answers:
                    value = answers[qid]
                    # 反向计分处理
                    if qid in rule.reverse_ids:
                        value = self._reverse_score(value)
                    dim_score += value
            
            # 归一化处理（转为百分制）
            max_score = len(rule.question_ids) * 5  # 假设每题最高5分
            scores[rule.dimension] = round(dim_score / max_score * 100, 1) * rule.weight
        
        return scores
    
    @staticmethod
    def _reverse_score(value: int, max_value: int = 5) -> int:
        return max_value + 1 - value
```

### 2.4 专业匹配算法

根据各维度得分，计算用户与 40 个专业的匹配度。

```python
# services/matching.py
import numpy as np
from typing import List, Tuple

@dataclass
class Profession:
    """专业定义"""
    id: int
    name: str                    # 专业名称
    dimension_weights: Dict[str, float]  # 各维度权重配置
    description: str             # 专业描述
    responsibilities: List[str]  # 相关工作职责

class ProfessionMatcher:
    """专业匹配引擎"""
    
    def __init__(self, professions: List[Profession]):
        self.professions = professions
    
    def match(
        self, 
        user_scores: Dict[str, float],
        top_n: int = 40
    ) -> List[Tuple[Profession, float]]:
        """
        计算用户与各专业的匹配度
        
        Returns:
            [(profession, match_score), ...] 按匹配度降序排列
        """
        results = []
        
        for prof in self.professions:
            # 余弦相似度计算
            prof_vector = []
            user_vector = []
            
            for dim in user_scores.keys():
                prof_vector.append(prof.dimension_weights.get(dim, 0))
                user_vector.append(user_scores[dim] / 100)  # 归一化
            
            similarity = self._cosine_similarity(
                np.array(prof_vector), 
                np.array(user_vector)
            )
            
            # 转换为百分比
            match_percentage = round(max(0, similarity) * 100, 1)
            results.append((prof, match_percentage))
        
        # 按匹配度降序排序
        results.sort(key=lambda x: x[1], reverse=True)
        return results[:top_n]
    
    @staticmethod
    def _cosine_similarity(a: np.ndarray, b: np.ndarray) -> float:
        dot = np.dot(a, b)
        norm_a = np.linalg.norm(a)
        norm_b = np.linalg.norm(b)
        if norm_a == 0 or norm_b == 0:
            return 0
        return dot / (norm_a * norm_b)
```

### 3.1 核心表结构

```sql
-- 用户表
CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    openid VARCHAR(64) UNIQUE NOT NULL,
    unionid VARCHAR(64),
    nickname VARCHAR(100),
    avatar_url TEXT,
    created_at TIMESTAMP DEFAULT NOW()
);

-- 量表表（支持多个量表）
CREATE TABLE scales (
    id SERIAL PRIMARY KEY,
    name VARCHAR(200) NOT NULL,        -- 量表名称
    description TEXT,                   -- 量表简介
    total_questions INT DEFAULT 0,      -- 题目总数
    estimated_time INT,                 -- 预计完成时间（分钟）
    status VARCHAR(20) DEFAULT 'active',
    created_at TIMESTAMP DEFAULT NOW()
);

-- 题目表
CREATE TABLE questions (
    id SERIAL PRIMARY KEY,
    scale_id INT REFERENCES scales(id),
    question_number INT NOT NULL,       -- 题号
    type VARCHAR(20) DEFAULT 'single',  -- single/multiple/text
    stem TEXT NOT NULL,                 -- 题干
    options JSONB,                      -- 选项：[{"label":"A","value":1,"text":"..."}]
    dimension VARCHAR(100),             -- 所属维度
    is_reverse BOOLEAN DEFAULT FALSE,   -- 是否反向计分
    UNIQUE(scale_id, question_number)
);

-- 测评结果表
CREATE TABLE results (
    id SERIAL PRIMARY KEY,
    user_id INT REFERENCES users(id),
    scale_id INT REFERENCES scales(id),
    answers JSONB NOT NULL,             -- 答题记录
    dimension_scores JSONB,             -- 各维度得分
    profession_ranking JSONB,           -- 专业匹配排序
    responsibility_analysis TEXT,       -- 工作职责分析
    completed_at TIMESTAMP DEFAULT NOW()
);

-- 专业表（40个专业配置）
CREATE TABLE professions (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    dimension_weights JSONB NOT NULL,   -- {"逻辑天赋":0.8,"艺术天赋":0.3,...}
    description TEXT,
    responsibilities TEXT[],            -- 工作职责列表
    category VARCHAR(50)                -- 专业大类
);
```

### 3.2 关键设计要点

- **JSONB 灵活存储**：题目的选项、结果中的得分和排名使用 JSONB 字段，方便扩展不同量表的个性化需求
- **维度可配置**：题目关联维度字段，评分规则可按量表独立配置
- **专业权重可调**：每个专业对各维度的权重可灵活设置，支持后续调整优化

### 6.3 关键注意事项

1. **小程序审核**：量表类小程序属于「教育—在线教育」或「工具—信息查询」类目，确保选择正确的类目避免审核被拒
2. **数据安全**：用户测评数据属于敏感信息，确保 HTTPS 传输，数据库访问权限严格控制
3. **性能优化**：量表题目较多时使用虚拟列表渲染，分包加载减少首屏加载时间
4. **离线容错**：答题过程中网络异常时，本地暂存答案，恢复后自动提交
5. **微信支付**：如需付费测评，对接微信支付接口（需开通商户号）
