import math


class ScoringEngine:
    LIKERT_MAP = {
        "A": 5,
        "B": 4,
        "C": 3,
        "D": 2,
        "E": 1,
        "a": 5,
        "b": 4,
        "c": 3,
        "d": 2,
        "e": 1,
    }

    def __init__(self, questions: list[dict], dimensions: list[dict]):
        self.questions = {q["id"]: q for q in questions}
        self.dimensions = {d["code"]: d for d in dimensions}
        self._dimension_index = {d["id"]: d for d in dimensions}

    def score_answers(self, answers: list[dict]) -> dict:
        raw_scores: dict[str, float] = {}
        question_counts: dict[str, int] = {}

        for dim in self.dimensions.values():
            raw_scores[dim["code"]] = 0.0
            question_counts[dim["code"]] = 0

        for answer in answers:
            qid = answer["question_id"]
            selected = answer["selected"]
            question = self.questions.get(qid)
            if not question:
                continue

            dim = self._dimension_index.get(question["dimension_id"])
            if not dim:
                continue

            raw_score = self.LIKERT_MAP.get(selected, 3)
            if question.get("is_reverse"):
                raw_score = 6 - raw_score

            dim_code = dim["code"]
            raw_scores[dim_code] = raw_scores.get(dim_code, 0.0) + raw_score
            question_counts[dim_code] = question_counts.get(dim_code, 0) + 1

        normalized_scores: dict[str, float] = {}
        for code, total in raw_scores.items():
            count = question_counts.get(code, 0)
            if count > 0:
                max_possible = count * 5
                normalized = (total / max_possible) * 100
                normalized_scores[code] = round(normalized, 2)
            else:
                normalized_scores[code] = 0.0

        rolled_up_scores = self._rollup_hierarchy(normalized_scores)
        return rolled_up_scores

    def _rollup_hierarchy(self, leaf_scores: dict[str, float]) -> dict[str, float]:
        result = dict(leaf_scores)

        dims_by_level = {1: [], 2: [], 3: []}
        for dim in self.dimensions.values():
            dims_by_level[dim["level"]].append(dim)

        level_3_parents: dict[str, list[str]] = {}
        for dim in dims_by_level[3]:
            parent = dim.get("parent_code")
            if parent:
                if parent not in level_3_parents:
                    level_3_parents[parent] = []
                level_3_parents[parent].append(dim["code"])

        for parent_code, child_codes in level_3_parents.items():
            child_scores = [result.get(c, 0.0) for c in child_codes if c in result]
            if child_scores:
                result[parent_code] = round(sum(child_scores) / len(child_scores), 2)

        level_2_parents: dict[str, list[str]] = {}
        for dim in dims_by_level[2]:
            parent = dim.get("parent_code")
            if parent:
                if parent not in level_2_parents:
                    level_2_parents[parent] = []
                level_2_parents[parent].append(dim["code"])

        for parent_code, child_codes in level_2_parents.items():
            child_scores = [result.get(c, 0.0) for c in child_codes if c in result]
            if child_scores:
                result[parent_code] = round(sum(child_scores) / len(child_scores), 2)

        return result

import math


class ProfessionMatcher:
    def __init__(self, professions: list[dict]):
        self.professions = professions

    def match(self, dimension_scores: dict[str, float], top_n: int = 10) -> list[dict]:
        results = []
        for prof in self.professions:
            weights = prof.get("dimension_weights", {})
            similarity = self._cosine_similarity(dimension_scores, weights)
            results.append({
                "profession_id": prof["id"],
                "profession_name": prof["name"],
                "category": prof.get("category"),
                "match_percentage": round(similarity * 100, 2),
                "description": prof.get("description"),
                "responsibilities": prof.get("responsibilities", []),
            })

        results.sort(key=lambda x: x["match_percentage"], reverse=True)
        return results[:top_n]

    def _cosine_similarity(self, vec_a: dict[str, float], vec_b: dict[str, float]) -> float:
        all_keys = set(vec_a.keys()) | set(vec_b.keys())

        dot_product = 0.0
        norm_a = 0.0
        norm_b = 0.0

        for key in all_keys:
            a_val = vec_a.get(key, 0.0)
            b_val = vec_b.get(key, 0.0)
            dot_product += a_val * b_val
            norm_a += a_val * a_val
            norm_b += b_val * b_val

        if norm_a == 0.0 or norm_b == 0.0:
            return 0.0

        return dot_product / (math.sqrt(norm_a) * math.sqrt(norm_b))

        ]

    def _identify_weaknesses(self, scores: dict[str, float]) -> list[str]:
        return [
            code for code, score in scores.items()

        seen = set()
        aggregated = []
        for prof in top_professions[:5]:
            for resp in prof.get("responsibilities", []):
                if resp not in seen:
                    seen.add(resp)
                    aggregated.append(resp)
        return aggregated[:10]

async def _wechat_code_to_openid(code: str) -> str | None:
    if not settings.WX_APPID or not settings.WX_SECRET:
        return f"dev_openid_{code}"

    url = "https://api.weixin.qq.com/sns/jscode2session"
    params = {
        "appid": settings.WX_APPID,
        "secret": settings.WX_SECRET,
        "js_code": code,
        "grant_type": "authorization_code",
    }

    try:
        async with httpx.AsyncClient(timeout=10.0) as client:
            resp = await client.get(url, params=params)
            resp.raise_for_status()
            data = resp.json()
            return data.get("openid")
    except Exception:
        return None


def _create_jwt_token(user_id: str, openid: str) -> str:
    expire = datetime.now(timezone.utc) + timedelta(minutes=settings.JWT_EXPIRE_MINUTES)
    payload = {
        "sub": user_id,
        "openid": openid,
        "exp": expire,
        "iat": datetime.now(timezone.utc),
    }
    return jwt.encode(payload, settings.JWT_SECRET, algorithm=settings.JWT_ALGORITHM)


def get_current_user_id(token: str) -> str:
    try:
        payload = jwt.decode(token, settings.JWT_SECRET, algorithms=[settings.JWT_ALGORITHM])
        return payload.get("sub", "")
    except Exception:
        return ""


async def require_user(authorization: str | None = None) -> str:
    if not authorization:
        raise HTTPException(status_code=401, detail="Missing authorization header")
    token = authorization.replace("Bearer ", "").replace("bearer ", "")
    user_id = get_current_user_id(token)
    if not user_id:
        raise HTTPException(status_code=401, detail="Invalid or expired token")
    return user_id
